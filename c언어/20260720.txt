
#include <stdio.h>

int main() {
	/* 배열과 포인터는 비슷해요
	배열하면 무조건 heap 영역에 공간 할당하고,
	거기에 시작주소를 ar1에 담는거에요 */
	int ar1[3] = {1,4,6};
	// 주소가 튀어 나와요. 왜냐면 ar1은 시작주소를 담고 있으니까요
	printf("ar1: %d \n", ar1); 

	// ar1이 저장하고 있는 주소 + 0을 한다음 거기 원산지 찾아가기
	printf("ar1[2]: %d \n", ar1[2]);

	for (int i = 0; i < 3;i++) {
		printf("ar1[%d]=%d \n", i, ar1[i]);
	}

	return 0;
}

#include <stdio.h>

int main() {
	char c1 = 'a';
	char c2 = 'a';
	char str1[10] = { 'h','e','l','l','o','\0'};
	char str2[10] = { 'h','e','l','l','o','\0'};
	char str3[10] = "hello";

	printf("c1 이랑 c2랑 같니? %d \n", c1 == c2); // 1

	/* 이건 str1이 가지고 있는 주소랑 str2가 가지고 있는 주소를 
	비교한것. 배열을 선언하면 무조건 새로운 공간에 공간을 만들기
	때문에 주소는 틀림 */
	printf("str1 이랑 str2랑 같니? %d \n", str1 == str2); // 0

	for (int i = 0; i < 10;i++) {
		if (str3[i] == '\0') break;
		printf("%c", str3[i]);
	}
	

	return 0;
}

#include <stdio.h>

int main() {
	int ar1[3][2] = {
		{1,2},
		{3,4},
		{5,6}
	};
	printf("ar1: %p \n", ar1);
	printf("ar1[0]: %p \n", ar1[0]);
	printf("ar1[1]: %p \n", ar1[1]);
	printf("ar1[2]: %p \n", ar1[2]);
	printf("ar1[2][0]: %d \n", ar1[2][0]);
	printf("*(*(ar1+2)+0): %d \n", *(*(ar1+2)+0));

	return 0;
}
