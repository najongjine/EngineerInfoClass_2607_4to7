#include <stdio.h>

typedef struct Person {
    int money;
    int id;
}Person;

int main() {
    /* 그냥 기본자료형에서 {{ 두번 나오면 2차원 배열 맞아요 */
    int ar1[2][2] = {
        {1,2},
        {3,4}
    };
    Person p1 = { 100,1 };
    // 주소복사가 아닌 값 복붙
    Person p2 = p1;
    // p2 라는구조체 안에서 id 라는변수 가져와
    p2.id = 2;

    /* struct 에선 배열에 {{ 이게 두번 나오는데, 2차원 배열 아니에요. 혼동하시면 안되요 */
    Person par1[2] = {
        {100,1},
        {200,2}
    };
    // {200,2} 이쪽 데이터를 말하는 거임
    par1[1].id;


    Person* pp1 = &p1;
    // 구조체 포인터에서 원산지 찾아갈땐 -> 걸로 써요
    pp1->id;

    int a = 1;
    int* pa = &a;
    // 기본자료형 원산지 찾아갈땐 * 써요
    *pa ;


    return 0;
}
