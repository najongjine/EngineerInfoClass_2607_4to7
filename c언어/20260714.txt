#include <stdio.h>

int main() {
    int a = 1;
    int b = 2;
    /* 정수/정수 := 정수 */
    int c = a / b;

    /* 정수/소수  ||  소수/소수 := 소수 
    (double) 이건 강제 형변환 */
    double c2 = a / (double)b;
    
    a = 1;
    b = 3;
    c = a % b;
    

    a = 3;
    /* shift 연산. 얘를 계산하기 위해선
    2진수로 변환 해야되요 
    왼쪽으로 밀땐 0이 오른쪽에 채워지고,
    오른쪽으로 밀땐 오른쪽끝에 낭떠러지가 있다 생각 하세요 */
    a = a << 1;
    a = a >> 2;

    a = 1;
    b = 2;
    c = a > b;
    c = a >= b;
    c = a == b; // 같니?
    // ! 는 반전.  true를 false, false를 true로
    c = a != b; // 같지 않니?
    /* 0, NULL, \0 := false
    나머지는 다 true */
    printf("c: %d \n", !(1>2));

    a = 1;
    b = 2;
    // AND 비트 연산: 2진수에서 하나라도 0이면 0
    c = a & b;
    // OR 비트 연산. 둘다 0이여야 0. 하나라도 1이면 1
    c = a | b;
    // XOR 비트 연산. 둘다 틀려야 1. 같으면 0
    c = a ^ b;

    a = 1;
    b = 0;
    c = a && b;
    c = a || b;

    c = 1;
    // c= c + 2;
    c += 2;
    
    return 0;
}
