#include <stdio.h>

int main() {
    int score = 99;
    char grade = 'F';
    /* 
    A : 90~100
    B : 80~89
    C : 70~79
    D : 60~69
    나머진 F
    */
    if (score >= 90 && score <=100) {
        grade = 'A';
    }
    else if (score >= 80 && score <= 89) {
        grade = 'B';
    }
    else if (score >= 70 && score <= 79) {
        grade = 'C';
    }
    else if (score >= 60 && score <= 69) {
        grade = 'D';
    }
    else {
        grade = 'F';
    }
    printf("grade : %c \n", grade);
    printf("끝 \n");
    return 0;
}

#include <stdio.h>

int main() {
    int a = 3;
    /* switch (matching)
    똑같은거 찾아 가는놈
    첫번째로 똑같은놈 찾고, 
    break 만나면 멈추고 빠져나옴
    break 없으면 계속 진행 */
    switch (a) {
        case 3:
            printf("3 \n");
            
        case 2:
            printf("2 \n");
            break;
        case 1:
            printf("1 \n");
            break;
        default:
            printf("몰라 \n");
    }

    printf("끝 \n");
    return 0;
}

#include <stdio.h>

int main() {
    /* 반복문
    조건이 true면 블록안에 있는 코드를 반복수행
    조건이 false면 빠져나옴 */
    /* 반복문의 종류
    while()
    do while()
    for()
    재귀함수
    */

    /* 반복문은 꼭 종료 조건이 있어야 한다. 안그러면
    무한루프에 빠져서 프로그램이 폭주하고 터진다 */
    
    int a = 0;

    /* 반복문의 종료조건을 만드는 방법
    1. 조건문 수학 수식을 잘 만들기
    2. break 걸기 */
    while (0) {
        printf("반복중 %d \n", ++a);
    }
    printf("끝 \n");
    return 0;
}
