#include <stdio.h>


int main() {
    /* 내장함수 := c언어 만든 사람이 이미 만들어준놈
    printf 의 기능 := 흑백 화면에 보여주기 */
    // \n := 다음줄 개행
    printf("ㅎㅇ \n");
    printf("오늘은 덥네요 \n");
    /* \ := 특수문자를 표현할때 사용되는놈
    \\ 하면 화면에서 \게 표현이 된다 */
    printf(" \\o/ \n");
    printf("\"덥다\" \n");

    // %d = 정수를 10진수로 보여줌
    printf("나는 퍼센트d, %d \n", 1);

    printf("퍼센트x, %x \n", 10);

    printf("퍼센트lf, %lf \n", 1.1);

    // %c := 문자 하나
    printf("퍼센트c, %c \n", 'A');

    /* 컴퓨터는 글자를 몰라서, 글자 하나하나를 이미지로
    저장해놓고, 각 글씨 하나당 숫자 id를 붙여요 
    이걸 전문용어로 아스키코드라 합니다 
    A = 65
    a = 97 */
    printf("%d \n", 'A');
    printf("%c \n", 65);

    int a = 1;
    int b = 2;
    printf("a:%d, b:%d \n", a, b);

    return 0;
}
