#include <stdio.h>


int main() {
    int i = 0;

    /* i가 0 ~ 3 까지 하고, 4가되면 빠져나와요 */
    while (i<=3) {
        printf("i: %d \n", i);
        i++;
    }
    while (1) {
        // 이건 조건체크에서 i가 4랑 같거나 보다크면 강제로 빠져나오기
        if (i >= 4) {
            break;
        }
        printf("i: %d \n", i);
        i++;
    }
    printf("마지막 i:%d \n", i);
    return 0;
}

#include <stdio.h>

int main() {
    int i = 0;

    /* do while은 조건체크하는 곳이 아래에 있어서
    한번은 무조건 수행하게 되있어요. 그다음 조건체크 해서 더 할지 안할지
    판단해요 */
    do {
        printf("저는 한번은 무조건 되요\n");
        i++;
    } while (i<3);

    printf("마지막 i:%d \n", i);
    return 0;
}

#include <stdio.h>

int main() {
    /* 
    * 1. for
    * 2. while
    * 3. do while
    */

    int i = 0;
    
    /*
    * int i=0 := 초기화 부분. 최초 한번만 실행
    * 1. i < 3 := 조건체크 부분
    * 2. { 내용물 }
    * 3. i++ 
    * for loop에 int i 라는걸 초기화 부분에 선언했으면, 이건
    * 새로운 scope에 변수따로 만든거라서, main scope에 있는거랑
    * 혼동하시면 안됩니다.
    */
    for (int i = 0; i < 3;i++) {
        printf("i: %d \n", i);
    }

    printf("끝 i: %d \n", i);
    return 0;
}
