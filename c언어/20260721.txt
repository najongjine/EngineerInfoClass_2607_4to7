#include <stdio.h>

int main(void) {
    int row0[2] = { 1, 2 };
    int row1[2] = { 3, 4 };
    int row2[2] = { 5, 6 };

    int* ar1[3] = {
        row0,
        row1,
        row2
    };

    ar1[0] = ar1[1];

    printf("%p\n", (void*)ar1[0]);
    printf("%p\n", (void*)ar1[1]);

    ar1[0][0] = 100;

    printf("%d\n", ar1[0][0]);
    printf("%d\n", ar1[1][0]);

    return 0;
}

#include <stdio.h>

int main(void) {
    int a = 1;
    int* pa = &a;
    int** ppa = &pa;
    int ar[2] = { 1,2 };

    printf("**ppa: %d \n", **ppa);
    printf("a:%d \n", a);
    printf("pa: %d \n", *pa);
    printf("pa: %d \n", pa[0]);

    printf("*ar=%d \n", *ar);
    printf("ar=%d \n", *(ar+1));
    printf("ar=%d \n", ar[1]);

    return 0;
}

#include <stdio.h>

int main(void) {
    /* 배열첨자 [] 가 나왔다 = 무조건 heap에 새로 깔아버려  */
    char str1[10] = "abc";
    /* 포인터 | String 타입으로 했다 = 처음 등장은 새로깔기
    그다음에 완전히 똑같은게 나오면 재사용 */
    char* str2 = "abc";
    char* str3 = "abc";

    printf("str1: %p \n", str1);
    printf("str2: %p \n", str2);
    printf("str3: %p \n", str3);

    return 0;
}
