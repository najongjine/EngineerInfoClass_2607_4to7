#include <stdio.h>
#include <stdlib.h>

int main() {
    /* 배열의 장점 : 쉬운편
    배열의 단점: 한번 크기가 정해지면 못바꿈 */
    int a[3] = { 1,2,3 };
    int a2[] = { 1,2,3 };

    int* par;
    /* malloc := heal 영역에 공간 새로 만들어 
    (sizeof(int)  여기를 잘 봐야해요.
    * 2 이런게 이으면 2칸 만들어.
    없으면 1칸만 만들어 */
    par = (int*)malloc(sizeof(int));
    printf("예전 par: %p \n", par);
    *par = 5;
    par[0] = 5;
    
    /* malloc으로 만든 공간의 크기를 변경하고 싶을때 쓰는놈
    * sizeof(int) * 3   새로운 공간 3개 새로 만들어. (예전공간 + 새로운 3 공간 아님)
    * 그 공간의 시작주소 뱉어서 par에 저장해
    * 예전 par 에 있던 데이터는 새로운 공간으로 이사시켜
    */
    par = (int*)realloc(par, sizeof(int) * 3);
    printf("realloc par: %p \n", par);
    printf("par:%d \n", *par);


    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int* func(int* arr, int size) {
    int* result = (int*)malloc(sizeof(int) * size);

    for (int i = 0; i < size; i++) {
        result[i] = *(arr + size - 1 - i) + i;
    }

    return result;
}

int main() {
    int arr[] = { 1, 2, 3, 4 };
    int length = sizeof(arr) / sizeof(arr[0]);
    int* result = func(arr, 4);

    for (int i = 0; i < 4; i++) {
        printf("%d ", result[i]);
    }

    free(result);

    return 0;
}
